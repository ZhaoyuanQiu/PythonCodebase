def vectorize_sequences(sequences,dimension=10000):

class NeuralNetwork(nn.Module):
    def __init__(self):
        super(NeuralNetwork, self).__init__()
        self.flatten = nn.Flatten()
        self.linear_relu_stack = nn.Sequential(
            nn.Linear(10000, 16),
            nn.ReLU(),
            nn.Linear(16, 16),
            nn.ReLU(),
            nn.Linear(16,2),
            nn.Sigmoid()
        )

    def forward(self, x):
        x = self.flatten(x)
        logits = self.linear_relu_stack(x)
        return logits

model = NeuralNetwork().to('cpu')
print(model)

NeuralNetwork(
  (flatten): Flatten(start_dim=1, end_dim=-1)
  (linear_relu_stack): Sequential(
    (0): Linear(in_features=10000, out_features=16, bias=True)
    (1): ReLU()
    (2): Linear(in_features=16, out_features=16, bias=True)
    (3): ReLU()
    (4): Linear(in_features=16, out_features=2, bias=True)
    (5): Sigmoid()
  )
)

def one_hot(x):
    tmp=[0.,0.]
    tmp[int(x.numpy())]=1.
    return tmp

def train(x_train,y_train, model, loss_fn, optimizer):
    #TELL MODEL IT SHOULD BE TRAINING
    model.train()

    #BATCH IMPLEMENTATION 
    X, y = torch.from_numpy(x_train).to(device).float(), torch.from_numpy(y_train).to(device).float()
    
    y=torch.tensor([one_hot(label) for label in y])
    # Compute prediction error
    pred = model(X)
    
    loss = loss_fn(pred, y)

    # Backpropagation
    optimizer.zero_grad()
    loss.backward()
    optimizer.step()

    # if t % 5 == 0:
    loss = loss.item() 
    train_loss_history.append(loss)

    if(i_print):
        print(f"loss: {loss:>7f}")

def validate(x_val,y_val, model, loss_fn):
    #TELL MODEL IT SHOULD BE EVALUATING
    model.eval()
    X, y = torch.from_numpy(x_val).to(device).float(), torch.from_numpy(y_val).to(device).float()
    y=torch.tensor([one_hot(label) for label in y])
    pred = model(X)

    val_loss = loss_fn(pred, y)
    val_loss_history.append(val_loss)
    if(i_print):
        print(f"Test Error: {val_loss:>8f}")

import torch
from torch import nn
from torchvision.transforms import ToTensor
device='cpu'

class NeuralNetwork(nn.Module):
    def __init__(self):
        super(NeuralNetwork, self).__init__()
        self.flatten = nn.Flatten()
        self.linear_relu_stack = nn.Sequential(
            nn.Linear(10000, 64),
            nn.ReLU(),
            nn.Linear(64, 64),
            nn.ReLU(),
            nn.Linear(64,46),
            nn.Softmax()
        )

    def forward(self, x):
        x = self.flatten(x)
        logits = self.linear_relu_stack(x)
        return logits

model = NeuralNetwork().to('cpu')
print(model)

    (3): ReLU()
    (4): Linear(in_features=64, out_features=46, bias=True)

def one_hot(x):
    tmp=[0.]*46
    tmp[int(x.numpy())]=1.
    return tmp

def train(x_train,y_train, model, loss_fn, optimizer):
    #TELL MODEL IT SHOULD BE TRAINING
    model.train()

    #BATCH IMPLEMENTATION 
    X, y = torch.from_numpy(x_train).to(device).float(), torch.from_numpy(y_train).to(device).float()
    
    y=torch.tensor([one_hot(label) for label in y])
    # Compute prediction error
    pred = model(X)
    
    loss = loss_fn(pred, y)

    # Backpropagation
    optimizer.zero_grad()
    loss.backward()
    optimizer.step()

    # if t % 5 == 0:
    loss = loss.item() 
    train_loss_history.append(loss)

    if(i_print):
        print(f"loss: {loss:>7f}")

def validate(x_val,y_val, model, loss_fn):
    #TELL MODEL IT SHOULD BE EVALUATING

    X, y = torch.from_numpy(x_val).to(device).float(), torch.from_numpy(y_val).to(device).float()
    y=torch.tensor([one_hot(label) for label in y])
    pred = model(X)

    val_loss = loss_fn(pred, y)
    val_loss_history.append(val_loss)
    if(i_print):
